# Abbreviations:
# ====================
#S = Singular
#P = Plural
#-P = Only with possessive suffix
#-------------------------
#Nom = Nominative (valo)
#Gen = Genetive (valon)
#Acc = Akkusative (valon, valo)
#Par = Partitive (valoa)
#Ine = Inessive (valossa)
#Ela = Elative (valosta)
#Ill = Illative (valoon)
#Ade = Adessive (valolla)
#Abl = Ablative (valolta)
#All = Allative (valolle)
#Ess = Essive (valona)
#Tra = Translative (valoksi)
#Ins = Instructive (valoin)
#Abe = Abessive (valotta)
#Com = Komitative (valoine-P)
#----------------------------
#G = Gradation
#S = Strong
#W = Weak
#---------------------
